from flask import Flask, render_template, request, redirect, url_for app = Flask(__name__) # Dummy data to simulate certificate storage certificates = [] admin_view = [] @app.route('/') def home(): return render_template('home.html') @app.route('/submit', methods=['GET', 'POST']) def submit(): if request.method == 'POST': name = request.form['name'] aadhar = request.form['aadhar'] reg = request.form['reg'] course = request.form['course'] category = request.form['category'] provider = request.form['provider'] link = request.form['link'] date = request.form['date'] if len(aadhar) == 12 and len(reg) >= 6: cert = { 'name': name, 'course': course, 'category': category, 'provider': provider, 'link': link, 'date': date } certificates.append(cert) admin_view.append(cert) return redirect(url_for('my_certificates')) else: return "Invalid Student Verification. Aadhar or Registration Number incorrect.", 400 return render_template('submit.html') @app.route('/my-certificates') def my_certificates(): return render_template('my_certificates.html', certificates=certificates) @app.route('/admin') def admin(): return render_template('admin.html', admin_data=admin_view) if __name__ == '__main__': app.run(debug=True)
